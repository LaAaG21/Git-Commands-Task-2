TASK 1:

START
	BEGIN
		SET NAME TO ''
		SET AGE TO ''

		DISPLAY 'Enter your name'
		READ NAME

		DISPLAY 'Enter your age'
		READ AGE

		DISPLAY 'Welcome to Codeline ' + NAME

		IF AGE GREATER THAN OR EQUALS TO 18 
		THEN DISPLAY 'Allowed'
		ELSE DISPLAY 'Not Allowed'
		END IF

	END
STOP
