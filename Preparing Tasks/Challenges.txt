Challenge 1:

START
	BEGIN
		DISPLAY 'Enter your age'
		READ AGE
		
		DISPLAY 'Are you a student?'
		READ BOOLEAN (IS_STUDENT)

		IF AGE LESS THAN 12
			THEN DISPLAY 'Child ticket: 2 OMR'

		ELSE IF AGE GRATER THAN OR EQUALSE TO 12 AND IS_STUDEN IS TRUE
			THEN DISPLAY 'Student ticket: 3 OMR'

		ELSE IF AGE GRATER THAN OR EQUALSE TO 60
			THEN DISPLAY 'Senior ticket: 5 OMR'

		ELSE DISPLAY 'Regular ticket: 5 OMR'

		END IF

		DISPLAY 'Thank you for visiting!'
	END
STOP


Challenge 2:

START
	BEGIN
		DISPLAY 'Enter your password length'
		READ NUMBER (PASSWORD_LENGTH)

			IF PASSWORD_LENGTH LESS THAN 6
				THEN DISPLAY 'Weak password — too short.'
				SET LENGTH_NEEDED TO 10 - Arithmetic (PASSWORD_LENGTH)
				DISPLAY 'You need to add ' + LENGTH_NEEDED + 'CHARACTERS TO THE PASSWORD'

			ELSE IF PASSWORD_LENGTH GRATER THAN OR EQUALSE TO 6 AND LESS THAN 10
				THEN DISPLAY 'Moderate password.'
				SET LENGTH_NEEDED TO 10 - Arithmetic (PASSWORD_LENGTH)
				DISPLAY 'You need to add ' + LENGTH_NEEDED + 'CHARACTERS TO THE PASSWORD'

			ELSE DISPLAY 'Strong password — well done!'
				DISPLAY 'No improvement needed.'

			END IF
		END REPEAT
	END
STOP


Challenge 3:


START
	BEGIN
		SET PASS_COUNT TO 0
		SET FAIL_COUNT TO 0

		REPEAT FROM NUM = 1 TO 5

			DISPLAY 'Enter your score'
			READ SCORE

			IF SCORE GRATER THAN OR EQUALS TO 90 AND LESS THAN 101
				THEN DISPLAY 'A'
				SET PASS_COUNT TO PASS_COUNT + 1

			ELSE IF SCORE GRATER THAN OR EQUALS TO 75 AND LESS THAN 90
				THEN DISPLAY 'B'
				SET PASS_COUNT TO PASS_COUNT + 1

			ELSE IF SCORE GRATER THAN OR EQUALS TO 60 AND LESS THAN 75
				THEN DISPLAY 'C'
				SET PASS_COUNT TO PASS_COUNT + 1

			ELSE IF SCORE GRATER THAN OR EQUALS TO 50 AND LESS THAN 60
				THEN DISPLAY 'D'
				SET PASS_COUNT TO PASS_COUNT + 1

			ELSE IF SCORE LESS THAN 50 AND GRATER THAN OR EQUALS TO 0
				THEN DISPLAY 'F'
				SET FAIL_COUNT TO FAIL_COUNT + 1
			
			ELSE DISPLAY 'WRONG SCORE. PLEASE ENTER BETWEEN 0 TO 100'
			END IF
		END REPEAT

		DISPLAY ' TOTAL STUDENTS WHO PASS ' + PASS_COUNT
		DISPLAY ' AND TOTAL STUDENTS WHO DID NOT PASS ' + FAIL_COUNT
	END
STOP


Challenge 4:

START
	BEGIN
		SET BALANCE TO 500
		SET WITHDRAWAL_COUNTER TO 0

		REPEAT WHILE WITHDRAWAL_COUNTER LEASS THAN 3 AND BALANCE GREATER THAN 0
			DISPLAY 'How much you want to withdraw?'
			READ AMOUNT

			IF AMOUNT GREATER THAN BALANCE
				THEN DISPLAY 'Insufficient funds. Try a smaller amount.'

			ELSE IF AMOUNT LESS THAN OR EQUALS 0
				THEN DISPLAY 'Invalid amount.'

			ELSE 
				SET BALANCE TO BALANCE - AMOUNT
				SET WITHDRAWAL_COUNTER TO  WITHDRAWAL_COUNTER + 1
				DISPLAY 'REMINING BALANCE ' + BALANCE
			END IF
		END REPEAT

		DISPLAY 'Session ended. Final balance: [$BALANCE]'
	END
STOP


Challenge 5:

START
	BEGIN
		SET SECRET TO 42
		SET ATTEMPTS TO 5
		SET IS_CURRECT TO FALSE

		REPEAT WHILE ATTEMPTS GREATER THAN 0 AND IS_CURRECT IS FALSE
			DISPLAY 'Attempts remining [$ATTEMPTS]'

			DISPLAY 'Guess the number'
			READ GUESS

			IF GUESS GREATER THAN SECRET
				THEN DISPLAY 'Too high! Try lower'

			ELSE IF GUESS LESS THAN SECRET
				THEN DISPLAY 'Too low! Try higher.'

			ELSE 
				DISPLAY 'Correct! You got it!'
				SET IS_CURRECT TO TRUE
			END IF
		END REPEAT

		IF IS_CURRECT IS FALSE AND ATTEMPTS EQUALSE TO 0
			THEN DISPLAY 'Game over. The number was 42'
	END
STOP


Challenge 6:

START
	BEGINE
		SET SANDWICH TO 'Sandwich'
		SET JUICE TO 'Juice'
		SET SALAD TO 'Salad'
		SET DESSERT TO 'Dessert'
		SET TOTAL TO 0

		REPEAT FROM NUM = 1 TO 4
			DISPLAY 'Sandwich = 1.2 OMR'
			DISPLAY 'Juice = 0.5 OMR'
			DISPLAY 'Salad = 0.8 OMR'
			DISPLAY 'Dessert = 1.0 OMR'

			DISPLAY 'Enter the item name'
			READ ITEM

			IF ITEM EQUALS TO SANDWICH
				THEN SET TOTAL TO TOTAL + 1.2

			ELSE IF ITEM EQUALS TO JUICE
				THEN SET TOTAL TO TOTAL + 0.5
		
			ELSE IF ITEM EQUALS TO SALAD
				THEN SET TOTAL TO TOTAL + 0.8

			ELSE IF ITEM EQUALS TO DESSERT
				THEN SET TOTAL TO TOTAL + 1.0

			ELSE DISPLAY 'Item not found.'
			END IF
		END REPEAT

		DISPLAY 'Total is [$TOTAL]'

		IF TOTAL GRATER THAN OR EQUALSE TO 3
			THEN DISPLAY 'The total after discount is [$TOTAL - ($TOTAL * 0.1)]'

		ELSE DISPLAY 'No discount applied.'
		END IF
	END
STOP
